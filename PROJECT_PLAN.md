#!/usr/bin/env python3
"""
Global News Fetcher
ä» RSS æºæŠ“å–æ–°é—»ï¼Œç”Ÿæˆ JSON æ•°æ®
"""

import feedparser
import json
from datetime import datetime, timezone
from pathlib import Path
import hashlib

def parse_rss_feed(url: str) -> list:
    """è§£æ RSS æº"""
    feed = feedparser.parse(url)
    articles = []
    
    for entry in feed.entries[:20]:  # æ¯ä¸ªæºå– 20 æ¡
        article = {
            'title': entry.title,
            'link': entry.link,
            'published': entry.get('published', ''),
            'source': feed.feed.get('title', 'Unknown'),
            'summary': entry.get('summary', '')[:200]  # é™åˆ¶æ‘˜è¦é•¿åº¦
        }
        articles.append(article)
    
    return articles

def deduplicate_articles(articles: list) -> list:
    """å»é‡ (åŸºäºæ ‡é¢˜å“ˆå¸Œ)"""
    seen = set()
    unique = []
    
    for article in articles:
        title_hash = hashlib.md5(article['title'].encode()).hexdigest()
        if title_hash not in seen:
            seen.add(title_hash)
            unique.append(article)
    
    return unique

def main():
    # åŠ è½½é…ç½®
    sources_file = Path('src/sources.json')
    with open(sources_file, 'r') as f:
        config = json.load(f)
    
    all_articles = []
    
    # æŠ“å–æ‰€æœ‰æº
    for source in config['sources']:
        print(f"Fetching {source['name']}...")
        articles = parse_rss_feed(source['rss'])
        
        # æ·»åŠ åˆ†ç±»ä¿¡æ¯
        for article in articles:
            article['category'] = source['category']
            article['country'] = source['country']
            article['language'] = source['language']
        
        all_articles.extend(articles)
    
    # å»é‡
    unique_articles = deduplicate_articles(all_articles)
    
    # æŒ‰æ—¶é—´æ’åº
    unique_articles.sort(
        key=lambda x: x.get('published', ''),
        reverse=True
    )
    
    # ç”Ÿæˆè¾“å‡º
    output = {
        'updated': datetime.now(timezone.utc).isoformat(),
        'total': len(unique_articles),
        'articles': unique_articles[:100]  # åªä¿ç•™æœ€æ–° 100 æ¡
    }
    
    # ä¿å­˜åˆ° data/news.json
    output_file = Path('data/news.json')
    output_file.parent.mkdir(exist_ok=True)
    
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump(output, f, indent=2, ensure_ascii=False)
    
    print(f"âœ… Saved {len(unique_articles)} articles to {output_file}")

if __name__ == '__main__':
    main()
